🚀 INITIAL SETUP (One-time)
─────────────────────────────────────────────────────────────────────────────
# 1. Create .env file with API credentials
cat > .env << 'EOF'
JAMF_URL=https://your-jamf-instance.jamfcloud.com
JAMF_CLIENT_ID=your_client_id
JAMF_CLIENT_SECRET=your_client_secret
SNIPE_IT_URL=https://your-snipe-it-instance.com
SNIPE_IT_API_TOKEN=your_api_token
EOF

# 2. Install Python dependencies
pip3 install -r requirements.txt


🧪 TEST CONNECTIONS (Recommended before first run)
─────────────────────────────────────────────────────────────────────────────
python3 test-connection.py


🎯 RUN THE SYNC (Primary Commands)
─────────────────────────────────────────────────────────────────────────────
# Interactive mode (recommended - with confirmations)
./run-ultimate-sync.sh

# Direct execution (no prompts)
python3 jamf-snipe-ultimate-100-percent-sync.py

# Background execution (for long-running sync)
nohup python3 jamf-snipe-ultimate-100-percent-sync.py > sync.log 2>&1 &


✅ VERIFY SYNC
─────────────────────────────────────────────────────────────────────────────
python3 verify-100-percent-sync.py
